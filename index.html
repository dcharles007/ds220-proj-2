import pandas as pd
import matplotlib.pyplot as plt
import re
df['cause_type'] = df['cause_of_death'].apply(lambda x: re.match(r"([a-zA-Z\s]+)", x).group(1) if pd.notnull(x) else '')
top_causes = df['cause_type'].value_counts()
top_causes.plot(kind='barh', title='Causes of Death', figsize=(15, 10))
plt.xlabel('Number of Deaths')
plt.ylabel('Cause')
plt.gca().invert_yaxis()  
plt.show()

df['year'] = df['yr_season'].str.extract(r'(\d{4})').astype(int)

import re
df['cause_type'] = df['cause_of_death'].apply(lambda x: re.match(r"([a-zA-Z\s]+)", x).group(1).strip() if pd.notnull(x) else '')
avg_age_by_cause = df.groupby('cause_type')['age'].mean().sort_values(ascending=False)
avg_age_by_cause.plot(kind='barh', title='Average Age by Cause of Death', figsize=(8, 5))
plt.xlabel('Average Age')
plt.ylabel('Cause of Death')
plt.gca().invert_yaxis()
plt.show()
